matrix identityMat {
    {1, 0, 0}
    {0, 1, 0}
    {0, 0, 1}
};


matrix t1 {
    {1, 2, 3}
    {4, 5, 6}
};

matrix t2 {
    {7, 8}
    {9, 10}
    {11, 12}
};


matrix v1 {
    {3, 2, 0, 1}
    {4, 0, 1, 2}
    {3, 0, 2, 1}
    {9, 2, 3, 1}
};

a = elemWisePow(elemWiseAdd(identityMat, 1), 2);
product = mult(t1, t2);
det = determinant(inverse(inverse(v1)));

print a, product, det;
export * >> all;
